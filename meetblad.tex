\section{Test-volgorde (kort)}
\begin{enumerate}[leftmargin=1.5em,itemsep=2pt,topsep=2pt]
\item Verbind app $\rightarrow$ lift, \textbf{laad de rail}. Druk \textbf{Nieuw logfile}; noteer logfile-naam hierboven.
\item \textbf{Baseline:} chair stil, [B] $\rightarrow$ $\geq$15\,s $\rightarrow$ [B].
\item \textbf{Frame-log [F] AAN} (voor DV3-smoothness op deze rail).
\item \textbf{B$\rightarrow$T:} rij in jog, stop bij elke marker, uitsettelen, dun streepje exact onder de pointer $\rightarrow$ [Spatie] (MARK). Noteer MP\# $\leftrightarrow$ marker.
\item \textbf{T$\rightarrow$B:} idem terug langs dezelfde markers.
\item \textbf{Frame-log [F] UIT.} Druk [S] (summary).
\item Achteraf: \texttt{adb pull} de log $\rightarrow$ analyse-script.
\end{enumerate}

{\footnotesize \textbf{Let op:} rail $>5$\,m $\Rightarrow$ accuratesse-drempel \textbf{10\,mm} (anders 5\,mm). \quad Gebruik een \textbf{dun streepje} als marker-referentie, niet de brede tape ($\sim$15\,mm onzekerheid).}
